// NaCasa Service Worker — Notificações de prazo const CACHE_NAME = 'nacasa-v1'; // Instala o service worker self.addEventListener('install', e => { self.skipWaiting(); }); self.addEventListener('activate', e => { e.waitUntil(clients.claim()); }); // Recebe mensagem do app para agendar verificação self.addEventListener('message', e => { if (e.data?.tipo === 'agendar-notificacoes') { agendarVerificacao(e.data.entregaveis); } }); // Verifica entregáveis com prazo amanhã e notifica function agendarVerificacao(entregaveis) { if (!entregaveis || !entregaveis.length) return; const hoje = new Date(); hoje.setHours(0, 0, 0, 0); const amanha = new Date(hoje); amanha.setDate(hoje.getDate() + 1); const amanhaStr = amanha.toISOString().split('T')[0]; const vencemAmanha = entregaveis.filter(d => d.prazo === amanhaStr && d.status !== 'entregue' ); if (vencemAmanha.length === 0) return; const nomes = vencemAmanha.map(d => d.ativ).join(', '); const titulo = vencemAmanha.length === 1 ? '⚠️ 1 entregável vence amanhã!' : `⚠️ ${vencemAmanha.length} entregáveis vencem amanhã!`; self.registration.showNotification(titulo, { body: nomes.length > 100 ? nomes.substring(0, 97) + '...' : nomes, icon: '/marketing-clientes/icon-192.png', badge: '/marketing-clientes/icon-192.png', tag: 'prazo-amanha', renotify: true, requireInteraction: true, actions: [ { action: 'abrir', title: 'Ver entregáveis' } ] }); } // Ao clicar na notificação, abre o app self.addEventListener('notificationclick', e => { e.notification.close(); e.waitUntil( clients.matchAll({ type: 'window', includeUncontrolled: true }).then(lista => { if (lista.length > 0) { lista[0].focus(); } else { clients.openWindow('/marketing-clientes/'); } }) ); });
